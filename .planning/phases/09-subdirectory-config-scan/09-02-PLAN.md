```typescript
export interface ScanResultEntry {
  configFileUri: vscode.Uri;
  dirUri: vscode.Uri;       // directory where config was found
  depth: number;
  configPriority: number;   // 0=behave.ini, 1=.behaverc, 2=setup.cfg, 3=tox.ini, 4=pyproject.toml
}

export interface ScanResult {
  primary: ScanResultEntry | undefined;       // first-match per D-11/D-12 ordering
  alsoFound: ScanResultEntry[];               // all other found configs (excludes primary)
  scannedDirs: number;                        // total directories visited
  circuitBreakerFired: boolean;               // true if maxEntriesScanned was reached
  maxDepthReached: number;                    // deepest level actually scanned
}
```

```typescript
const scanResultCache = new Map<string, ScanResult>();

export function getCachedScanResult(wkspUri: vscode.Uri): ScanResult | undefined {
  return scanResultCache.get(uriId(wkspUri));
}

export function setCachedScanResult(wkspUri: vscode.Uri, result: ScanResult): void {
  scanResultCache.set(uriId(wkspUri), result);
}

export function clearScanResultCache(): void {
  scanResultCache.clear();
}
```

**Main BFS scanner function:**
```typescript
export async function scanForBehaveConfig(
  wkspUri: vscode.Uri,
  maxDepth: number,
  stopOnFirstHit: boolean = false,
  maxEntriesScanned: number = 5000
): Promise<ScanResult>
```

Implementation requirements:
1. **If maxDepth === 0:** Return immediately with empty result (no scan performed). This is the v1.1 root-only behavior per SD-02.
2. **BFS traversal:** Use a queue of `{ uri: vscode.Uri, depth: number }`. Start by reading workspace root's immediate children at depth 1. Process directories level by level.
3. **Per-directory config check:** For each directory at depth 1..maxDepth, call `findBehaveConfig(dirUri)`. If it returns a result with `ok: true`, create a `ScanResultEntry` with the configFileUri from the result, dirUri, depth, and configPriority (look up the config filename in CONFIG_PRIORITY).
4. **Exclude directories:** Use `isDirExcluded` helper logic (replicate the function locally since it's not exported from common.ts, OR import DEFAULT_EXCLUDE_DIRS and replicate the exclusion check). Check: `if (DEFAULT_EXCLUDE_DIRS.has(dirName) || dirName.startsWith('.'))` — also skip hidden directories (starting with `.`) since behave configs never live there.
5. **Symlink cycle protection:** Maintain a `Set<string>` of visited real paths. Before entering a directory, resolve its real path via `fs.realpathSync.native()` wrapped in try-catch. If the realpath is already in the visited set, skip. This prevents infinite loops from circular symlinks.
6. **Circuit breaker:** Increment a counter for each directory entry read. If counter >= maxEntriesScanned, stop BFS, set `circuitBreakerFired = true` on the result.
7. **discoveryStopOnFirstHit:** If `stopOnFirstHit` is true and at least one config has been found at the current depth, do NOT enqueue directories at deeper levels. Continue processing the current depth queue to find all configs at this depth, then stop.
8. **Sorting / primary selection:** After BFS completes, sort all found entries by `(depth ASC, configPriority ASC)`. The first entry becomes `primary`. The rest go into `alsoFound`.
9. **Use `vscode.workspace.fs.readDirectory`** for directory listing (async, cross-platform). Filter entries to `vscode.FileType.Directory` for traversal. Do NOT recurse into files.
10. **Symlink detection:** `vscode.FileType.SymbolicLink` may be OR'd with `FileType.Directory`. Check `(fileType & vscode.FileType.Directory) !== 0` rather than strict equality.
11. **diagLog** calls for: scan start, each directory entered, each config found, circuit breaker fired, scan complete with summary.

Import `DEFAULT_EXCLUDE_DIRS` and `uriId` from `'../common'`. Import `findBehaveConfig` from `'../parsers/configParser'`. Import `diagLog` from `'../logger'`. Import `fs` for realpathSync. Import `vscode`.

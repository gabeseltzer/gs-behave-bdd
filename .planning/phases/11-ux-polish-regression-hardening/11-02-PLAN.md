   Set `discoveryDepth` to 0 via `vscode.workspace.getConfiguration().update()`, then trigger re-discovery:
   ```typescript
   test('discoveryDepth=0 disables subdirectory scanning', async function () {
     this.timeout(300000);
     // Set discoveryDepth to 0 — should restore v1.1 behavior (root-only)
     await vscode.workspace.getConfiguration('gs-behave-bdd').update(
       'discoveryDepth', 0, vscode.ConfigurationTarget.Workspace
     );
     
     // Wait for re-discovery to complete after settings change
     // The configurationChangedHandler fires on settings change
     // The monorepo-scan has no root-level config, so discoveryDepth=0 should yield no features
     const state = await waitForTestTree(
       () => {
         const entry = instances.getDiscoveryEntry(wkspUri);
         // After discoveryDepth=0, there should be no discovery entry OR entry with convention fallback
         // Since there's no root features/ dir either, the workspace should have no features
         if (!entry) return { noEntry: true };
         // If entry exists but has no features (convention fallback to nonexistent features/)
         // that's also acceptable — the key assertion is no subdirectory scan results
         if (entry.source === 'config-file' && (entry.configFileUri?.fsPath.includes('app-a') || entry.configFileUri?.fsPath.includes('app-b'))) {
           return undefined; // still seeing subdir config — not yet updated
         }
         return { entry };
       },
       { intervalMs: 100, timeoutMs: 15000 }
     );
     
     if ('noEntry' in state) {
       // No discovery entry at all — correct, no root config + no subdir scan
       assert.ok(true, 'no DiscoveryEntry when discoveryDepth=0 and no root config');
     } else {
       // Entry exists but should NOT be from a subdirectory config
       assert.notStrictEqual(state.entry!.source, 'config-file',
         'discoveryDepth=0 should not discover subdirectory configs');
     }
   });
   ```
